SELECT  escola.descricao AS nome_escola,  pontuacao.descricao AS pontuacao_desc, curso.descricao AS curso_desc, fase.descricao AS fase_desc, ano.descricao AS ano_desc, modalidade.descricao AS modalidade_desc, sexo.descricao AS sexo_desc FROM public.fato AS fato LEFT JOIN public.escola AS escola ON fato.escola = escola.id INNER JOIN public.pontuacao AS pontuacao ON fato.pontuacao = pontuacao.id INNER JOIN public.curso AS curso ON fato.curso = curso.id INNER  JOIN public.fase AS fase ON fato.fase = fase.id INNER  JOIN public.ano AS ano ON fato.ano = ano.id INNER  JOIN public.modalidade AS modalidade ON fato.modalidade = modalidade.id INNER  JOIN public.sexo AS sexo ON fato.sexo = sexo.id
